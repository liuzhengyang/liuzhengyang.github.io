# ForkJoinPool

# 设计选择

  • ForkJoinPool的work steal，是空闲Worker随机扫描其他WorkQueue实现的，这种吞吐通常比主动转移任务要高，因为线程唤醒需要时间（类似非公平锁）

# 核心数据结构

  • WorkQueue[] queues: 任务队列的数组，长度为2的n次方，queues数组中index为奇数的是Worker的WorkQueue，每个Worker有自己的WorkQueue，除此之外index为偶数的是submission queue。
  • WorkQueue: 任务队列，内部是Task数组，长度为2的n次方。

WorkQueue通过数组实现，数组长度为2的n次方。 外部提交任务给ForkJoinPool

submission queue: 从非forkjoinpool线程提交的任务会随机选择一个Work Queue来提交任务。

基于随机扫描(randomized scan)，比producer分配task给idle thread吞吐更高，因为可能有让完成任务的thread立刻接受此任务，避免了额外的线程的唤醒。

Worker会优先运行本地队列的任务（LIFO或FIFO模式，可以配置），如果本地没有任务，则会随机从其他的队列中steal。

Worker的创建是on-demand(按需)的，通过任务提交、替换已经结束的worker或者因为有worker被阻塞而补充等原因创建。

# 构造函数关键参数

  • parallelism: 并发度，默认情况使用CPU核数
  • asyncMode：如果是true，则使用先进先出的调度策略，适合非fork join的任务场景，VirtualThread使用的就是asyncMode为true
  • corePoolSize: 类似ThreadPoolExecutor中的corePoolSize，默认和parallelism一样。
  • maximumPoolSize：最多的Worker数量
  • keepAliveTime：worker数量超过corePoolSize之后，多余的worker会在keepAliveTime之后销毁。

# 创建流程

ForkJoinWorkerThread对象构造函数会创建一个 new ForkJoinPool.WorkQueue对象。 在线程启动后，run方法的运行会调用ForkJoinPool的registerWorker注册Worker，然后调用ForkJoinPool的runWorker运行Worker。

class ForkJoinWorkerThread {
    ForkJoinWorkerThread(ThreadGroup group, ForkJoinPool pool,
                         boolean useSystemClassLoader,
                         boolean clearThreadLocals) {
        super(group, null, pool.nextWorkerThreadName(), 0L, !clearThreadLocals);
        UncaughtExceptionHandler handler = (this.pool = pool).ueh;
        this.workQueue = new ForkJoinPool.WorkQueue(this, 0);
        if (clearThreadLocals)
            workQueue.setClearThreadLocals();
        super.setDaemon(true);
        if (handler != null)
            super.setUncaughtExceptionHandler(handler);
        if (useSystemClassLoader)
            super.setContextClassLoader(ClassLoader.getSystemClassLoader());
    }

    public void run() {
        Throwable exception = null;
        ForkJoinPool p = pool;
        ForkJoinPool.WorkQueue w = workQueue;
        if (p != null && w != null) {   // skip on failed initialization
            try {
                p.registerWorker(w);
                onStart();
                p.runWorker(w);
            } catch (Throwable ex) {
                exception = ex;
            } finally {
                try {
                    onTermination(exception);
                } catch (Throwable ex) {
                    if (exception == null)
                        exception = ex;
                } finally {
                    p.deregisterWorker(this, exception);
                }
            }
        }
    }
}

注册Worker会对WorkQueue进行初始化创建数组，并且设置到ForkJoinPool的queues中，设置的index是偶数，如果queues满了会进行扩容。

class ForkJoinPool {
    final void registerWorker(WorkQueue w) {
        ThreadLocalRandom.localInit();
        int seed = ThreadLocalRandom.getProbe();
        ReentrantLock lock = registrationLock;
        int cfg = config & FIFO;
        if (w != null && lock != null) {
            w.array = new ForkJoinTask<?>[INITIAL_QUEUE_CAPACITY];
            cfg |= w.config | SRC;
            w.stackPred = seed;
            int id = (seed << 1) | 1;                   // initial index guess
            lock.lock();
            try {
                WorkQueue[] qs; int n;                  // find queue index
                if ((qs = queues) != null && (n = qs.length) > 0) {
                    int k = n, m = n - 1;
                    for (; qs[id &= m] != null && k > 0; id -= 2, k -= 2);
                    if (k == 0)
                        id = n | 1;                     // resize below
                    w.phase = w.config = id | cfg;      // now publishable

                    if (id < n)
                        qs[id] = w;
                    else {                              // expand array
                        int an = n << 1, am = an - 1;
                        WorkQueue[] as = new WorkQueue[an];
                        as[id & am] = w;
                        for (int j = 1; j < n; j += 2)
                            as[j] = qs[j];
                        for (int j = 0; j < n; j += 2) {
                            WorkQueue q;
                            if ((q = qs[j]) != null)    // shared queues may move
                                as[q.config & am] = q;
                        }
                        U.storeFence();                 // fill before publish
                        queues = as;
                    }
                }
            } finally {
                lock.unlock();
            }
        }
    }
}

# runWorker

runWorker是Worker线程的取任务逻辑，是一个循环。 每次回先调用scan(w, src, r)，scan会返回。 如果scan返回的为负数，则会调用awaitWork等待任务。 下一次循环时，会对r进行修改以便获取下一个随机数。

class ForkJoinPool {
    final void runWorker(WorkQueue w) {
        if (w != null) {                        // skip on failed init
            int r = w.stackPred, src = 0;       // use seed from registerWorker
            do {
                r ^= r << 13;
                r ^= r >>> 17;
                r ^= r << 5; // xorshift
            } while ((src = scan(w, src, r)) >= 0 ||
                    (src = awaitWork(w)) == 0);
            w.access = STOP;                    // record normal termination
        }
    }
}

class ForkJoinPool {
    private int scan(WorkQueue w, int prevSrc, int r) {
        WorkQueue[] qs = queues;
        int n = (w == null || qs == null) ? 0 : qs.length;
        for (int step = (r >>> 16) | 1, i = n; i > 0; --i, r += step) {
            int j, cap; WorkQueue q; ForkJoinTask<?>[] a;
            if ((q = qs[j = r & (n - 1)]) != null &&
                    (a = q.array) != null && (cap = a.length) > 0) {
                int src = j | SRC, b = q.base;
                int k = (cap - 1) & b, nb = b + 1, nk = (cap - 1) & nb;
                ForkJoinTask<?> t = a[k];
                U.loadFence();                  // for re-reads
                if (q.base != b)                // inconsistent
                    return prevSrc;
                else if (t != null && WorkQueue.casSlotToNull(a, k, t)) {
                    q.base = nb;
                    w.source = src;
                    if (src + (src << SWIDTH) != prevSrc &&
                            q.base == nb && a[nk] != null)
                        signalWork();           // propagate at most twice/run
                    w.topLevelExec(t, q);
                    return src + (prevSrc << SWIDTH);
                }
                else if (q.array != a || a[k] != null || a[nk] != null)
                    return prevSrc;             // revisit
            }
        }
        return -1;
    }
}

class ForkJoinPool {

    private int awaitWork(WorkQueue w) {
        if (w == null)
            return -1;                           // currently impossible
        int p = (w.phase + SS_SEQ) & ~INACTIVE;  // advance phase
        boolean idle = false;                    // true if possibly quiescent
        if (runState < 0)
            return -1;                           // terminating
        long sp = p & SP_MASK, pc = ctl, qc;
        w.phase = p | INACTIVE;
        do {                                     // enqueue
            w.stackPred = (int)pc;               // set ctl stack link
        } while (pc != (pc = compareAndExchangeCtl(
                pc, qc = ((pc - RC_UNIT) & UC_MASK) | sp)));
        if ((qc & RC_MASK) <= 0L) {
            if (hasTasks(true) && (w.phase >= 0 || reactivate() == w))
                return 0;                        // check for stragglers
            if (runState != 0 && tryTerminate(false, false))
                return -1;                       // quiescent termination
            idle = true;
        }
        WorkQueue[] qs = queues; // spin for expected #accesses in scan+signal
        int spins = ((qs == null) ? 0 : ((qs.length & SMASK) << 1)) | 0xf;
        while ((p = w.phase) < 0 && --spins > 0)
            Thread.onSpinWait();
        if (p < 0) {
            long deadline = idle ? keepAlive + System.currentTimeMillis() : 0L;
            LockSupport.setCurrentBlocker(this);
            for (;;) {                           // await signal or termination
                if (runState < 0)
                    return -1;
                w.access = PARKED;               // enable unpark
                if (w.phase < 0) {
                    if (idle)
                        LockSupport.parkUntil(deadline);
                    else
                        LockSupport.park();
                }
                w.access = 0;                    // disable unpark
                if (w.phase >= 0) {
                    LockSupport.setCurrentBlocker(null);
                    break;
                }
                Thread.interrupted();            // clear status for next park
                if (idle) {                      // check for idle timeout
                    if (deadline - System.currentTimeMillis() < TIMEOUT_SLOP) {
                        if (tryTrim(w))
                            return -1;
                        else                     // not at head; restart timer
                            deadline += keepAlive;
                    }
                }
            }
        }
        return 0;
    }
}

# WorkQueue

核心字段

  • int stackPred: 线程池中的栈，
  • int base: poll时使用的index
  • ForkJoinTask<?>[] array: 任务队列，长度是2的n次方
  • final ForkJoinWorkerThread owner: 拥有这个WorkQueue的Worker，或者null的情况下表示是共享的submission队列
  • int top: push时使用的index
  • int access: 出入队时使用的锁
static final class WorkQueue {
    int stackPred;             // pool stack (ctl) predecessor link
    int config;                // index, mode, ORed with SRC after init
    int base;                  // index of next slot for poll
    ForkJoinTask<?>[] array;   // the queued tasks; power of 2 size
    final ForkJoinWorkerThread owner; // owning thread or null if shared

    // fields otherwise causing more unnecessary false-sharing cache misses
    @jdk.internal.vm.annotation.Contended("w")
    int top;                   // index of next slot for push
    @jdk.internal.vm.annotation.Contended("w")
    volatile int access;       // values 0, 1 (locked), PARKED, STOP
    @jdk.internal.vm.annotation.Contended("w")
    volatile int phase;        // versioned, negative if inactive
    @jdk.internal.vm.annotation.Contended("w")
    volatile int source;       // source queue id in topLevelExec
    @jdk.internal.vm.annotation.Contended("w")
    int nsteals;               // number of steals from other queues

}

class WorkQueue {
    final void push(ForkJoinTask<?> task, ForkJoinPool pool,
                    boolean signalIfEmpty) {
        boolean resize = false;
        int s = top++, b = base, cap, m; ForkJoinTask<?>[] a;
        if ((a = array) != null && (cap = a.length) > 0) {
            if ((m = (cap - 1)) == s - b) {
                resize = true;            // rapidly grow until large
                int newCap = (cap < 1 << 24) ? cap << 2 : cap << 1;
                ForkJoinTask<?>[] newArray;
                try {
                    newArray = new ForkJoinTask<?>[newCap];
                } catch (Throwable ex) {
                    top = s;
                    access = 0;
                    throw new RejectedExecutionException(
                            "Queue capacity exceeded");
                }
                if (newCap > 0) {         // always true
                    int newMask = newCap - 1, k = s;
                    do {                  // poll old, push to new
                        newArray[k-- & newMask] = task;
                    } while ((task = getAndClearSlot(a, k & m)) != null);
                }
                array = newArray;
            }
            else
                a[m & s] = task;
            getAndSetAccess(0);           // for memory effects if owned
            if ((resize || (a[m & (s - 1)] == null && signalIfEmpty)) &&
                    pool != null)
                pool.signalWork();
        }
    }
}

class WorkQueue {
    final ForkJoinTask<?> poll(ForkJoinPool pool) {
        for (int b = base;;) {
            int cap; ForkJoinTask<?>[] a;
            if ((a = array) == null || (cap = a.length) <= 0)
                break;                        // currently impossible
            int k = (cap - 1) & b, nb = b + 1, nk = (cap - 1) & nb;
            ForkJoinTask<?> t = a[k];
            U.loadFence();                    // for re-reads
            if (b != (b = base))              // inconsistent
                ;
            else if (t != null && casSlotToNull(a, k, t)) {
                base = nb;
                U.storeFence();
                if (pool != null && a[nk] != null)
                    pool.signalWork();        // propagate
                return t;
            }
            else if (array != a || a[k] != null)
                ;                             // stale
            else if (a[nk] == null && top - b <= 0)
                break;                        // empty
        }
        return null;
    }
}

# Task生产流程

  • 提交会判断提交线程，如果是ForkJoinWorkerThread，则会提交到当前线程的WorkQueue中。
  • 如果不是，则通过externalPush随机找到一个WorkQueue入队。
private &lt;T> ForkJoinTask&lt;T> externalSubmit(ForkJoinTask&lt;T> task) {
    Thread t; ForkJoinWorkerThread w; WorkQueue q;
    if (task == null)
        throw new NullPointerException();
    if (((t = Thread.currentThread()) instanceof ForkJoinWorkerThread) &amp;&amp;
        (w = (ForkJoinWorkerThread)t).pool == this &amp;&amp;
        (q = w.workQueue) != null)
        q.push(task);
    else
        externalPush(task);
    return task;
}

push方法细节

  • Worker线程的WorkQueue是私有队列，只有当前线程进行入队，所以不需要考虑并发问题（在可见性方面加了一个fullFence保证其他Worker能看到并steal）
  • base指向队头，top指向队尾，入队则将task设置到top为止并给top++
  • WorkQueue数组长度是2的n次方，可以通过&(长度-1)实现取模运算
  • 入队时检查如果当前队列长度为0，则入队后要调用ForkJoinPool.singnalWork唤醒Worker来工作（如果没有足够Worker的情况）
  • 如果队列满了，则进行扩容，扩容后数组长度为之前的两倍。

    ForkJoinTask&lt;?>[] a;

        top = s + 1;
        if (((d = s - (int)BASE.getAcquire(this)) &amp; ~1) == 0 &amp;&amp;
            p != null) {                 // size 0 or 1
            VarHandle.fullFence();
            p.signalWork();
        }
        else if (d == m)
            growArray(false);
    }
}

externalPush

externalPush会提交到共享的submission queue，submission queue是workQueues数组中index为偶数的WorkQueue。 入队前会通过CAS加锁，入队完成解锁。如果该index的WorkQueue为空，则会尝试创建WorkQueue。

final void externalPush(ForkJoinTask&lt;?> task) {
    int r;                                // initialize caller's probe
    if ((r = ThreadLocalRandom.getProbe()) == 0) {
        ThreadLocalRandom.localInit();
        r = ThreadLocalRandom.getProbe();
    }
    for (;;) {
        WorkQueue q;
        int md = mode, n;
        WorkQueue[] ws = workQueues;
        if ((md &amp; SHUTDOWN) != 0 || ws == null || (n = ws.length) &lt;= 0)
            throw new RejectedExecutionException();
        else if ((q = ws[(n - 1) &amp; r &amp; SQMASK]) == null) { // add queue
            int qid = (r | QUIET) &amp; ~(FIFO | OWNED);
            Object lock = workerNamePrefix;
            ForkJoinTask&lt;?>[] qa =
                new ForkJoinTask&lt;?>[INITIAL_QUEUE_CAPACITY];
            q = new WorkQueue(this, null);
            q.array = qa;
            q.id = qid;
            q.source = QUIET;
            if (lock != null) {     // unless disabled, lock pool to install
                synchronized (lock) {
                    WorkQueue[] vs; int i, vn;
                    if ((vs = workQueues) != null &amp;&amp; (vn = vs.length) > 0 &amp;&amp;
                        vs[i = qid &amp; (vn - 1) &amp; SQMASK] == null)
                        vs[i] = q;  // else another thread already installed
                }
            }
        }
        else if (!q.tryLockPhase()) // move if busy
            r = ThreadLocalRandom.advanceProbe(r);
        else {
            if (q.lockedPush(task))
                signalWork();
            return;
        }
    }
}

# signalWork

ctl是ForkJoinPool中存储复合信息的long字段 64个bit按照16bit一组分别记录不同的信息

  • 最高位的16个bit：released worker减去目标parallism的值
  • 次高位的16个bit：所有的worker减去target parallism的值
  • 次低位的16个bit：版本号和栈顶worker的状态
  • 最低位的16个bit：栈顶worker的index

任务队列：通过treiber stack结构维护，ctl中存储了栈顶的worker index，唤醒时唤醒栈顶worker。 这种方式能够按照最近使用原则（最近刚要park的worker），提升性能和局部性。 如何防止single miss问题(任何一个wait/signal设计都需要考虑的），available worker在入队之后需要再重新scan task。 没有scan到后，available worker就会block知道被signalled。

  • released worker：正在运行或扫描任务的worker
  • unreleased/available worker: 在ctl stack中记录的worker。
  • WorkQueue通过stackPred形成stack，便于快速找到空闲Worker

signalWork的实现步骤是

  • 先判断ctl是否>=0，即判断最高为的16个bit的值，也就是released worker和target parallism，如果release worker大于等于target parallism，不需要唤醒新的worker。
  • 否则在看ctl的低32位，如果是0，说明没有等待的worker。再通过c & ADD_WORKER判断次高位的16bit，如果不等于0说明也是负数，需要增加worker，调用addWorker创建一个新Worker并返回。
  • 然后通过sp & SMASK获取ctl的低16位值，即栈顶worker的index
  • 最后需要唤醒栈顶worker，唤醒前，要从栈顶worker的stackPred（出栈之后下一个worker的index）作为新的ctl的低16位的值进行cas修改ctl，修改成功后unpark唤醒。
final void signalWork() {
    for (;;) {
        long c; int sp; WorkQueue[] ws; int i; WorkQueue v;
        if ((c = ctl) >= 0L)                      // enough workers
            break;
        else if ((sp = (int)c) == 0) {            // no idle workers
            if ((c &amp; ADD_WORKER) != 0L)           // too few workers
                tryAddWorker(c);
            break;
        }
        else if ((ws = workQueues) == null)
            break;                                // unstarted/terminated
        else if (ws.length &lt;= (i = sp &amp; SMASK))
            break;                                // terminated
        else if ((v = ws[i]) == null)
            break;                                // terminating
        else {
            int np = sp &amp; ~UNSIGNALLED;
            int vp = v.phase;
            long nc = (v.stackPred &amp; SP_MASK) | (UC_MASK &amp; (c + RC_UNIT));
            Thread vt = v.owner;
            if (sp == vp &amp;&amp; CTL.compareAndSet(this, c, nc)) {
                v.phase = np;
                if (vt != null &amp;&amp; v.source &lt; 0)
                    LockSupport.unpark(vt);
                break;
            }
        }
    }
}

# Worker消费流程

  • 生成一个随机数r，从这个r开始遍历workQueues，如果找到有不为空的WorkQueue，则尝试steal
  • steal成功后，调用topLevelExec，执行这个任务。
  • 执行完成后，会持续执行自己Worker的WorkQueue，直到为空后再去刚steal到WorkQueue去poll任务，
  • 如果都没有任务了，则返回再换一个随机数重复刚才的流程，知道都没有任务了，则需要park阻塞等待新任务。
private boolean scan(WorkQueue w, int r) {
    WorkQueue[] ws; int n;
    if ((ws = workQueues) != null &amp;&amp; (n = ws.length) > 0 &amp;&amp; w != null) {
        for (int m = n - 1, j = r &amp; m;;) {
            WorkQueue q; int b;
            if ((q = ws[j]) != null &amp;&amp; q.top != (b = q.base)) {
                int qid = q.id;
                ForkJoinTask&lt;?>[] a; int cap, k; ForkJoinTask&lt;?> t;
                if ((a = q.array) != null &amp;&amp; (cap = a.length) > 0) {
                    t = (ForkJoinTask&lt;?>)QA.getAcquire(a, k = (cap - 1) &amp; b);
                    if (q.base == b++ &amp;&amp; t != null &amp;&amp;
                        QA.compareAndSet(a, k, t, null)) {
                        q.base = b;
                        w.source = qid;
                        if (q.top - b > 0)
                            signalWork();
                        w.topLevelExec(t, q,  // random fairness bound
                                       r &amp; ((n &lt;&lt; TOP_BOUND_SHIFT) - 1));
                    }
                }
                return true;
            }
            else if (--n > 0)
                j = (j + 1) &amp; m;
            else
                break;
        }
    }
    return false;
}

runWorker不断scan，知道scan不出来任务。计算新的phase值（负数表示block），并设置自己的stackPred为当前ctl的低32位。 设置完成后再scan一遍防止signal信号并发丢失。 再scan完成后，如果phase没有被修改说明没有被唤醒，则可以放心park，进入最后的park流程。

WorkQueue的phase字段含义

  • 如果是负数，表示处于block状态
  • 0标识可运行状态
  • 正数标识版本号
final void runWorker(WorkQueue w) {
    int r = (w.id ^ ThreadLocalRandom.nextSecondarySeed()) | FIFO; // rng
    w.array = new ForkJoinTask&lt;?>[INITIAL_QUEUE_CAPACITY]; // initialize
    for (;;) {
        int phase;
        if (scan(w, r)) {                     // scan until apparently empty
            r ^= r &lt;&lt; 13; r ^= r >>> 17; r ^= r &lt;&lt; 5; // move (xorshift)
        }
        else if ((phase = w.phase) >= 0) {    // enqueue, then rescan
            long np = (w.phase = (phase + SS_SEQ) | UNSIGNALLED) &amp; SP_MASK;
            long c, nc;
            do {
                w.stackPred = (int)(c = ctl);
                nc = ((c - RC_UNIT) &amp; UC_MASK) | np;
            } while (!CTL.weakCompareAndSet(this, c, nc));
        }
        else {                                // already queued
            int pred = w.stackPred;
            Thread.interrupted();             // clear before park
            w.source = DORMANT;               // enable signal
            long c = ctl;
            int md = mode, rc = (md &amp; SMASK) + (int)(c >> RC_SHIFT);
            if (md &lt; 0)                       // terminating
                break;
            else if (rc &lt;= 0 &amp;&amp; (md &amp; SHUTDOWN) != 0 &amp;&amp;
                     tryTerminate(false, false))
                break;                        // quiescent shutdown
            else if (rc &lt;= 0 &amp;&amp; pred != 0 &amp;&amp; phase == (int)c) {
                long nc = (UC_MASK &amp; (c - TC_UNIT)) | (SP_MASK &amp; pred);
                long d = keepAlive + System.currentTimeMillis();
                LockSupport.parkUntil(this, d);
                if (ctl == c &amp;&amp;               // drop on timeout if all idle
                    d - System.currentTimeMillis() &lt;= TIMEOUT_SLOP &amp;&amp;
                    CTL.compareAndSet(this, c, nc)) {
                    w.phase = QUIET;
                    break;
                }
            }
            else if (w.phase &lt; 0)
                LockSupport.park(this);       // OK if spuriously woken
            w.source = 0;                     // disable signal
        }
    }
}
